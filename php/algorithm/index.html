function longestCommonSubstring($a, $b){
    //还是同样的问题，PHP如何解决二维数组的问题

    $len_a = strlen($a);
    $len_b = strlen($b);
    //初始化一个二维数组  ---> php中如何初始化二维数组找了好久
    $dp_a = array_fill(0,$len_b+1,0);
    $dp = array_fill(0,$len_a+1, $dp_a);
    //动态规划的思路，dp[i][j]代表一个状态，可以由之前的状态来得到
    for($i = 0; $i < $len_a; $i++){
        for ($j = 0; $j < $len_b; $j++) {
            //开始状态转移
            if (substr($a, $i, 1) === substr($b, $j, 1)) {
//                $dp[$i]= dp[$i - 1]+ 1;
                $dp[$i+1][$j+1] = $dp[$i][$j]+1;
            }else{
                $dp[$i+1][$j+1] = max($dp[$i][$j+1], $dp[$i+1][$j]);
            }
        }
    }
    return $dp[$len_a][$len_b];
//    return ans;
}
