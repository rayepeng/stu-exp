        public static int find(int array[], int key){
        int start = 0, end = (array.length - 1);

        while(start <= end && key >= array[start] && key <= array[end]){
            int pos = start + ((end - start) /(array[end] - array[start]) * (key - array[start]));

            if(array[pos] == key){
                return pos;
            }if(array[pos] < key){
                start = pos + 1;
            }else{
                end = pos - 1;
            }
        }
        return -1;
    }

        public TreeNode find(int value){
        TreeNode current = Root;
        while(current != null){
            if(value < current.val){
                if(current.left == null){
                    return null;
                }
                current = current.left;
            }else if(value > current.val){
                if(current.right == null){
                    return null;
                }
                current = current.right;
            }else{
                return current;
            }
        }
        return null;
    }

         public int ternarySearch(Integer[] arr, Integer key, int start, int end){
        if(start > end){
            return -1;
        }
        int mid1 = start + (end - start) /3;
        int mid2 = start + 2*(end - start)/3;

        if(key.compareTo(arr[mid1]) == 0){
            return mid1;
        }else if(key.compareTo(arr[mid2]) == 0){
            return mid2;
        }
        else if(key.compareTo(arr[mid1]) < 0){
            return ternarySearch(arr, key, start, --mid1);
        }else if(key.compareTo(arr[mid2]) > 0){
            return ternarySearch(arr, key, ++mid2, end);
        }
        else{
            return ternarySearch(arr, key, mid1, mid2);
        }

    }

    这个算法其实也要假定数据是排好序的

    //寻找第k个最小的元素   
public int quickSelect(int[] nums, int k, int start, int end) {

        //随机选择一个数
        Random rand = new Random();
        int index = rand.nextInt(end - start + 1) + start;
        int pivot = nums[index];
        swap(nums, index, end);

        int left = start, right = end;

        while(left < right) {
            if (nums[left++] >= pivot) {
                swap(nums, --left, --right);
            }
        }
        //left is now pointing to the first number that is greater than or equal to the pivot
        swap(nums, left, end);

        //m 代表有多少个小于pivot的数
        int m = left - start;

        if (m == k - 1) { //寻找第k个最小的数
            return pivot;
        }
        else if (k <= m) { //目标在左边的子数列
            return quickSelect(nums, k, start, left - 1);
        }
        else { 
            //目标在右边的子数列，但是要更新k       
            return quickSelect(nums, k - m, left, end);
        }
    }

    算法的描述：

    从数组选择一个属作为pivot,交换pivot和数组最后一个数的位置

    随机的pivot有利于算法运行在平均的时间复杂度

    将数组分成两半，左边的都小于等于pivot右边的都大于等于pivot

    然后k与左边的数组长度比较，如果

    1. k = len + 1 pivot就是要找的
    2. k <= len 从左边数组递归
    3. k > len 更新k = k -len 从右边数组递归
